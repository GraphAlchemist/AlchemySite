        var config = {
            dataSource: 'data/smallmammal.json',
            forceLocked: false,
            nodeTypes: {"type": ["class","species","family","order"]},
            nodeStyle: {
              "class":{
                "color": "#008080",
                "borderColor": "#008080"
              },
              "order": {
                "color": "#ff7373",
                "borderColor": "#ff7373"

              },
              "family": {
                "color": "#008000",
                "borderColor": "#008000"
              },
              "species": {
                "color": "#800000",
                "borderColor": "#800000"
              },
            edgeStyle: {
              "all":{
                "color": function(d){

                  var len = d.getProperties().source.length;
                  var pos = len % 2;

                  var hex = [len,len,len];
                  hex[pos] = len > 9 ? "FF" : "F";
                  hex[2] = len > 9 ? "00" : len;

                  var color = "#" + hex.join("");

                  return color;
                },
                opacity:0.4
              }
            }
            };

        alchemy = new Alchemy(config)
    </script>
